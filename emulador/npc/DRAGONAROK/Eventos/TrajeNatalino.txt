prontera,156,167,5	script	Roupas Natalinas	714,{

mes "[ Garota ]";
mes "Olá!";
if (RoupaNatal == 1) goto L_Fantasiado;
	mes "Deseja colocar a roupa de natal por 5 minutos?";
	menu "Sim",-,"Não",L_Cancel;
		sc_start SC_Xmas,600000,0;
		initnpctimer;
		attachnpctimer strcharinfo(0);
		set RoupaNatal, 1;
		close;

L_Fantasiado:
	mes "Deseja tirar a roupa de natal?";
	menu "Sim",-,"Não",L_Cancel;
	sc_end SC_Xmas;
	set RoupaNatal, 0;
	close;

L_Cancel:
	next;
	mes "[ Garota ]";
	mes "Até a próxima...";
	close;

OnTimer0:
	set RoupaNatal, 0;
	detachnpctimer;
	stopnpctimer;
	end;
}
