-	script	DayNight	-1,{
	OnClock0600:
		day;
		announce "O Sol se levanta sobre toda Rune-Midgard...",0;
		end;
				
	OnClock1200:
		announce "Mais uma tarde no horizonte, o Sol está em seu ponto mais alto...",0;
		end;
		
	OnClock1900:
		night;
		announce "Anoitece e os habitantes de Rune-Midgard retornam para suas casas...",0;
		end;
		
	OnClock0000:
		announce "A Madrugada toma conta de Rune-Midgard.",0;
		end;
	
	
	OnInit:
	if(gettime(3)>=6 && gettime(3)<19){
	
	goto DIA;
}
	else
{
	goto NOITE;
}
	DIA:
	day;
	announce "Está de Dia.",0;
	
	NOITE:
	night;
	announce "Está de Noite.",0;
	
}
