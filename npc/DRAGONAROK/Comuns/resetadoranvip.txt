prontera,166,230,4	script	Resetinha	4_F_TELEPORTER,{

	set .@ResetStat,5000000;	// Zeny requeridos para resetar atributos.
	set .@ResetSkill,5000000;	// Zeny requeridos para resetar habilidades.
	set .@ResetBoth,9000000;	// Zeny requeridos para resetar ambos.

	mes "[Resetinha]";
	mes "Olá "+strcharinfo(0)+", eu sou ^FF3355Belladora^000000, a Resetinha.";
	mes "Resetar Atributos: "+ .@ResetStat +"z";
	mes "Resetar Habilidades: "+ .@ResetSkill +"z";
	mes "Resetar Ambos: "+ .@ResetBoth +"z";
	mes "Por favor, selecione o serviço que deseja:";
	next;
	mes "SELECIONE SUA OPÇÃO:";
	switch(select("^FF3355Resetar Habilidades:Resetar Atributos:Resetar Ambos^000000:Cancelar")) {
	case 1:
		mes "[Resetinha]";
		if (Zeny < .@ResetSkill) {
			mes "Desculpe-me, mas você não possui zeny suficiente para esta tarefa.";
			close;
		}
		Zeny -= .@ResetSkill;
		sc_end SC_ALL;
		resetskill;
		mes "Pronto Atributos resetados, volte quando precisar de algum serviço.";
		close;
	case 2:
		mes "[Resetinha]";
		if (Zeny < .@ResetStat) {
			mes "Desculpe-me, mas você não possui zeny suficiente para esta tarefa.";
			close;
		}
		Zeny -= .@ResetStat;
		resetstatus;
		mes "Pronto Habilidades resetadas, volte quando precisar de algum serviço.";
		close;
	case 3:
		mes "[Resetinha]";
		if (Zeny < .@ResetBoth) {
			mes "Desculpe-me, mas você não possui zeny suficiente para esta tarefa.";
			close;
		}
		Zeny -= .@ResetBoth;
		sc_end SC_ALL;
		resetskill;
		resetstatus;
		mes "Pronto seus Atributos e suas Habilidades foram resetadas, volte quando precisar de algum serviço.";
		close;
	case 4:
		mes "[Resetinha]";
		mes "Tudo bem, volte quando precisar de algum serviço.";
		close;
	}
	OnInit:
waitingroom "Resetadora",0;
end;
}
